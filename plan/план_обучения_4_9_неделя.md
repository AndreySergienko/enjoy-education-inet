Тестовое задание на синке:(сам когда-то тренировался на таком)

class SafeEmitter<Events extends Record<string, any>> {
private handlers: {
[K in keyof Events]?: ((payload: Events[K]) => void)[]
} = {};

    on<K extends keyof Events>(
        event: K,
        handler: (payload: Events[K]) => void
    ): void {
        (this.handlers[event] ??= []).push(handler);
    }

    emit<K extends keyof Events>(
        event: K,
        payload: Events[K]
    ): void {
        this.handlers[event]?.forEach(h => h(payload));
    }

}

const emitter = new SafeEmitter<EventMap>();

emitter.on('login', payload => {
payload.userId; // string — всё ок
});

emitter.emit('login', { userId: '123' }); // ✔
emitter.emit('login', { foo: 'bar' }); // ❌ TS-ошибка
emitter.emit('unknown', { }); // ❌ TS-ошибка
